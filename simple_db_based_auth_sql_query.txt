create table "Admin_Data" (
  id bigint generated by default as identity primary key,
  email text not null,
  password text not null
);

-- 1. Enable Row Level Security on the Admin_Data table.
-- This is the master switch. Once enabled, all access is denied by default
-- until you create policies to allow specific actions.
ALTER TABLE "Admin_Data" ENABLE ROW LEVEL SECURITY;

-- 2. Create a policy to allow public read access.
-- This policy is necessary for your login function to work. It allows the
-- database to check for a matching email and password. Because your client-side
-- query is very specific, it doesn't expose the whole table.
CREATE POLICY "Allow public read access for login check"
ON "public"."Admin_Data" FOR SELECT
USING (true);

-- 3. Create a policy to allow updates.
-- IMPORTANT: This policy allows any authenticated user to ATTEMPT an update.
-- The real security for this action comes from your application code, where you
-- correctly verify the user's `currentPassword` before sending the update.
-- This is less secure than using Supabase's built-in auth but fits your current model.
CREATE POLICY "Allow credential updates via application logic"
ON "public"."Admin_Data" FOR UPDATE
USING (true)
WITH CHECK (true);

-- 4. Create policies to explicitly DENY inserts and deletes from the client.
-- For a simple admin system, you should manage users directly in the Supabase
-- dashboard, not allow them to be created or deleted from the web app.
CREATE POLICY "Block all client-side inserts"
ON "public"."Admin_Data" FOR INSERT
WITH CHECK (false);

CREATE POLICY "Block all client-side deletes"
ON "public"."Admin_Data" FOR DELETE
USING (false);
